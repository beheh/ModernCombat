RelaunchPosition

Kategorie: Calls

Beschreibung

Wird aufgerufen, wenn Relaunchpositionen festgestellt werden müssen, zB bei den Hazard-Zielen und Hold The Flag.
Die Funktion bekommt zwei Referenzen übergeben, die als X- und Y-Position gesetzt werden müssen.

Syntax

public func RelaunchPosition (int & iX, int & iY, int iTeam)

Beispiel

public func RelaunchPosition(int &iX, int &iY, int iTeam)
{
  //DM, LMS, CTF und HTF haben dieselben Spawnpunkte
  if (FindObject(GTDM) || FindObject(GLMS) || FindObject(GCTF) || FindObject(GHTF))
    //Team 1
    if (iTeam == 1)
      //Zufälliger Spawnpunkt
      if (!Random(2))
      {
        iX = 150;
        iY = 150;
        return true;
      }
      else
      {
        iX = 100;
        iY = 300;
        return true;
      }
    //Team 2
    else
      //Zufälliger Spawnpunkt
      if (!Random(2))
      {
        iX = 1350;
        iY = 150;
        return true;
      }
      else
      {
        iX = 1400;
        iY = 300;
        return true;
      }

  //Besitznahme und Assaults verwalten Spawnpunkte selbst, also wird die Startsicht abgefragt
  iX = LandscapeWidth() / 2;
  iY = LandscapeHeight() / 2;
  return true;
}
